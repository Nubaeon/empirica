Core Classes

CanonicalEpistemicCascade

The main class for running complete epistemic reasoning workflows.

class CanonicalEpistemicCascade:
    """Complete CASCADE workflow implementation"""

    def __init__(
        self,
        # Profile-based configuration (recommended)
        profile_name: Optional[str] = None,
        ai_model: Optional[str] = None,
        domain: Optional[str] = None,

        # Agent configuration
        agent_id: str = "cascade",

        # Feature toggles
        enable_bayesian: bool = True,
        enable_drift_monitor: bool = True,
        enable_session_db: bool = True,
        enable_git_notes: bool = True,
        enable_action_hooks: bool = True,
        auto_start_dashboard: bool = False,

        # Advanced options
        investigation_plugins: Optional[Dict[str, InvestigationPlugin]] = None,
        enable_perspective_caching: bool = True,
        cache_ttl: int = 300,
        session_id: Optional[str] = None
    ):
        """Initialize CASCADE with profile-based configuration"""

    async def run_epistemic_cascade(
        self, 
        task: str,
        context: Optional[Dict[str, Any]] = None
    ) -> Dict[str, Any]:
        """
        Execute complete CASCADE workflow

        Phases:
        1. PREFLIGHT - Baseline epistemic assessment
        2. THINK - Initial reasoning
        3. PLAN - Task breakdown (internal)
        4. INVESTIGATE - Fill knowledge gaps
        5. CHECK - Validate readiness
        6. ACT - Execute with confidence
        7. POSTFLIGHT - Measure learning

        Returns:
            Dict with keys:
            - action: str (Action.ACT, Action.INVESTIGATE, Action.CLARIFY)
            - confidence: float (0.0-1.0)
            - rationale: str
            - recommended_action: str
            - vector_summary: Dict[str, float]
            - calibration: Dict (PREFLIGHT vs POSTFLIGHT comparison)
            - epistemic_delta: Dict (learning measurements)
        """

Constructor Parameters:

Parameter Type Default Description
profile_name str None Investigation profile ("balanced", "autonomous_agent", "high_reasoning")
ai_model str None Auto-select profile by AI model ("claude", "gpt-4", "gemini")
domain str None Domain context for threshold tuning ("security", "research", "critical_domain")
agent_id str "cascade" Unique identifier for agent/session
enable_bayesian bool True Enable Bayesian Guardian belief tracking
enable_drift_monitor bool True Enable behavioral drift detection
enable_session_db bool True Enable SQLite session persistence
enable_git_notes bool True Enable git checkpoint integration (~450 tokens)
enable_action_hooks bool True Enable real-time tmux dashboard updates
auto_start_dashboard bool False Auto-launch tmux dashboard
investigation_plugins Dict None Custom investigation tools
enable_perspective_caching bool True Cache LLM perspectives
cache_ttl int 300 Cache time-to-live in seconds
session_id str None Explicit session ID for tracking (auto-generated if None)

Method: run_epistemic_cascade()

Parameter Type Required Description
task str Yes Task description to assess
context Dict No Execution context with keys like "files", "cwd", "requirements"

Returns: Dict with action recommendation, confidence, rationale, and epistemic measurements

Task & Context Format

Task: Simple string describing what needs to be done

task = "Review authentication system for SQL injection vulnerabilities"

Context: Dictionary with execution environment information

context = {
    "files": ["./auth/", "./security/"],           # Files to analyze
    "cwd": "/path/to/project",                      # Working directory
    "requirements": ["security_focus", "fast"],     # Task requirements
    "environment": {"runtime": "production"},       # Environment info
    "capabilities": ["file_read", "code_analysis"], # Available tools
    "history": [],                                   # Previous interactions
    "metadata": {}                                   # Additional context
}

Python API Components

CanonicalEpistemicAssessor

Low-level epistemic assessment (used internally by CASCADE).

from empirica.core.canonical import CanonicalEpistemicAssessor

assessor = CanonicalEpistemicAssessor(agent_id="my-agent")

# Assess epistemic state
assessment = assessor.assess_task(
    task_description="Review authentication security",
    context={"files": ["./auth/"]}
)

# Access assessment results
print(f"Engagement: {assessment.engagement.score:.2f}")
print(f"KNOW: {assessment.know.score:.2f}")
print(f"DO: {assessment.do.score:.2f}")
print(f"Overall Confidence: {assessment.overall_confidence:.2f}")
print(f"Engagement Gate Passed: {assessment.engagement_gate_passed}")

SessionDatabase

Persist and query epistemic sessions.

from empirica.data.session_database import SessionDatabase

db = SessionDatabase()

# Create session
session_id = db.create_session(
    ai_id="my-agent",
    bootstrap_level=2,
    components_loaded=5
)

# Log assessment
db.log_epistemic_assessment(
    cascade_id=session_id,
    assessment=assessment,
    phase="preflight"
)

# Query sessions
sessions = db.list_sessions(ai_id="my-agent", limit=10)

# Get session details
session = db.get_session(session_id)

db.close()

Best Practices

1. Use Profiles

# Good - profile-based
cascade = CanonicalEpistemicCascade(profile_name="autonomous_agent")

# Avoid - manual configuration
cascade = CanonicalEpistemicCascade(
    threshold_uncertainty_high=0.7,
    max_investigation_rounds=3,
    # ... many parameters
)

2. Enable Git Integration

cascade = CanonicalEpistemicCascade(
    enable_git_notes=True,
    enable_session_db=True
)

3. Use Handoffs for Multi-Session Work

# End of session
create_handoff_report(
    session_id=session_id,
    task_summary="Completed authentication review",
    key_findings=["OAuth2 implemented", "Tests passing"],
    next_context="Deploy to staging"
)

# Next session
resume_previous_session(ai_id="my-agent")

4. Leverage MCP Tools for AI Assistants

For AI assistants, always use MCP tools instead of direct Python API calls. They provide better integration and session management.
