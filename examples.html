Code Patterns and Real-World Workflows

1. Automated Code Review

Pattern: Epistemic filtering to reduce false positives.

from empirica import CanonicalEpistemicCascade

class EpistemicCodeReviewer:
    def __init__(self):
        self.cascade = CanonicalEpistemicCascade(
            agent_id="code-reviewer",
            enable_bayesian=True,
            enable_drift_monitor=True
        )

    async def review_code(self, code_files):
        task = f"Review {len(code_files)} files for security vulnerabilities"
        context = {
            "files": code_files,
            "domain": "code_analysis"
        }

        result = await self.cascade.run_epistemic_cascade(task, context)

        # Filter by confidence to reduce noise
        if result.get('overall_confidence', 0) > 0.8:
            print(f"✅ High confidence review: {result.get('rationale', 'N/A')}")
            return result
        else:
            print(f"⚠️ Low confidence ({result.get('overall_confidence', 0):.2f})")
            return None

2. Strategic Literature Review

Pattern: Systematic investigation with uncertainty tracking.

from empirica import CanonicalEpistemicCascade

async def review_literature(topic):
    # CASCADE handles PREFLIGHT/INVESTIGATE/CHECK/ACT/POSTFLIGHT internally
    cascade = CanonicalEpistemicCascade(
        agent_id="lit-reviewer",
        profile_name="autonomous_agent"  # More investigation rounds
    )

    task = f"Review literature on {topic} and identify key findings"
    context = {"domain": "research"}

    result = await cascade.run_epistemic_cascade(task, context)

    # Check learning delta
    if 'epistemic_delta' in result:
        delta = result['epistemic_delta']
        print(f"Knowledge gained: KNOW {delta.get('know', 0):+.2f}")
        print(f"Uncertainty reduced: {delta.get('uncertainty', 0):+.2f}")

    return result

3. Multi-AI Collaboration with Handoff Reports

Pattern: Using handoff reports for multi-agent workflows.

# Agent 1: Security Review
bootstrap_session(ai_id="security-agent", bootstrap_level=2)
execute_preflight(
    session_id="latest:active:security-agent",
    prompt="Review authentication system for vulnerabilities"
)
# ... perform security review ...

# Create handoff report for next agent
create_handoff_report(
    session_id="latest:active:security-agent",
    task_summary="Completed security review - found 2 issues",
    key_findings=["SQL injection risk", "Weak password hashing"],
    remaining_unknowns=["Impact on existing sessions"],
    next_session_context="Ready for performance review",
    artifacts_created=["security_report.md"]
)

# Agent 2: Performance Review (reads handoff)
bootstrap_session(ai_id="perf-agent", bootstrap_level=2)
handoffs = query_handoff_reports(ai_id="security-agent", limit=1)

4. CI/CD Gate Check Script

Pattern: Shell script integration using CLI.

#!/bin/bash

# Run assessment
RESULT=$(empirica assess --context ./src --output json)

# Parse confidence
CONFIDENCE=$(echo $RESULT | jq .confidence)

# Gate logic
if (( $(echo "$CONFIDENCE < 0.75" | bc -l) )); then
  echo "❌ Confidence too low for deployment."
  exit 1
else
  echo "✅ Deployment approved."
  exit 0
fi

5. Custom Investigation Plugin

Pattern: Extending Empirica with domain tools.

from empirica.core.metacognitive_cascade.investigation_plugin import InvestigationPlugin

class SecurityScannerPlugin(InvestigationPlugin):
    """Custom plugin for security scanning"""

    def __init__(self):
        super().__init__(
            name="security_scanner",
            description="Runs security scans on code",
            domains=["security", "code_analysis"]
        )

    async def investigate(self, knowledge_gap: str, context: dict) -> dict:
        """Execute security investigation"""
        if "security" in knowledge_gap.lower():
            # Run external security tool
            files = context.get('files', [])
            # scan_results = await run_security_scan(files)

            return {
                "findings": ["Example finding 1", "Example finding 2"],
                "confidence": 0.85,
                "sources": ["security_tool_output.json"]
            }

        return {"findings": [], "confidence": 0.0}

# Use plugin with CASCADE
cascade = CanonicalEpistemicCascade(
    agent_id="secure-reviewer",
    investigation_plugins={"security_scanner": SecurityScannerPlugin()}
)
