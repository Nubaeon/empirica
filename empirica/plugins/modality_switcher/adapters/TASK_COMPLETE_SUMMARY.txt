from modality_switcher.adapters import MinimaxAdapter
from empirica.core.modality.plugin_registry import AdapterPayload

# Initialize
adapter = MinimaxAdapter()

# Create payload
payload = AdapterPayload(
    system="You are a helpful assistant",
    state_summary="Testing adapter",
    user_query="What is 2+2?",
    temperature=0.2,
    max_tokens=100
)

# Make call
response = adapter.call(payload, {})

# Results
print(f"Decision: {response.decision}")           # ACT
print(f"Confidence: {response.confidence:.2f}")   # 0.70
print(f"Vectors: {len(response.vector_references)}") # 13
